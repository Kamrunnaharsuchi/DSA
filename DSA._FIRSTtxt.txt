#include<bits/stdc++.h>
using namespace std;

int main() {
    int arr[100];
    int n;

    cout << "Enter the size of array: ";
    cin >> n;

    cout << "Enter " << n << " numbers: ";
    for (int i = 0; i < n; i++) cin >> arr[i];

    int choice, pos, value;

    while (1) {

        cout << "1. INSERTION\n";
        cout << "2. DELETION\n";
        cout << "3. DISPLAY\n";
        cout << "4. SORT\n";
        cout << "5. SEARCH\n";
        cout << "6. EXIT\n";

        cout << "Enter your choice: ";
        cin >> choice;

        if (choice == 1) {
            cout << "Enter position: ";
            cin >> pos;
            cout << "Enter value to insert: ";
            cin >> value;

            if (pos < 0 || pos > n) {
                cout << "Invalid position!\n";
            } else {
                for (int i = n - 1; i >= pos; i--)
                    arr[i + 1] = arr[i];
                arr[pos] = value;
                n++;
                cout << "shabash!insert hoishee\n";
            }
        }

        else if (choice == 2) {
            cout << "Enter position to delete: ";
            cin >> pos;
            if (pos >= 0 && pos < n) {
                for (int i = pos; i < n - 1; i++)
                    arr[i] = arr[i + 1];
                n--;
                cout << "vai delete koira dilen!\n";
            } else {
                cout << "Invalid position!\n";
            }
        }

        else if (choice == 3) {
            cout << "Current Array: ";
            for (int i = 0; i < n; i++)
                cout << arr[i] << " ";
            cout << endl;
        }

        else if (choice == 4) {
            sort(arr, arr + n);
            cout << " congo!!!Array sorted!\n";
        }

        else if (choice == 5) {
            int searchChoice;
            cout << "1. Linear Search\n";
            cout << "2. Binary Search\n";
            cout << "Enter search choice: ";
            cin >> searchChoice;

            cout << "Enter value to search: ";
            cin >> value;

            bool found = false;
            int index = -1;

            if (searchChoice == 1) {
                for (int i = 0; i < n; i++) {
                    if (arr[i] == value) {
                        found = true;
                        index = i;
                        break;
                    }
                }
            }
            else if (searchChoice == 2) {
                sort(arr, arr + n);
                int low = 0, high = n - 1;
                while (low <= high) {
                    int mid = (low + high) / 2;
                    if (arr[mid] == value) {
                        found = true;
                        index = mid;
                        break;
                    } else if (arr[mid] < value)
                        low = mid + 1;
                    else
                        high = mid - 1;
                }
            }
            else {
                cout << "Invalid search choice!\n";
                continue;
            }

            if (found)
                cout << " Value found at index " << index << "!\n";
            else
                cout << "Value not found!\n";
        }

        else if (choice == 6) {
            cout << " Exiting program. Byeeeee!\n";
            break;
        }

        else {
            cout << "Invalid choice!abr den vai.\n";
        }
    }

    return 0;
}
